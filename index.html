async function loadGlobalHistory() { const status = document.getElementById('scan-status'); const grid = document.getElementById('master-grid'); grid.innerHTML = ""; try { // 1. Fetch TX list with a small retry if it fails const apiUrl = `https://api-sepolia.etherscan.io/api?module=account&action=txlist&address=${DEPLOYER}&sort=desc&apikey=FREE_TOKEN`; const response = await fetch(apiUrl); const data = await response.json(); if (data.status === "0" && data.message === "NOTOK") { status.innerText = "⏳ Rate limited. Retrying in 5 seconds..."; setTimeout(loadGlobalHistory, 5000); return; } // 2. Identify all Contract Creations const contracts = [...new Set(data.result .filter(tx => tx.to === "" || (tx.contractAddress && tx.contractAddress !== "")) .map(tx => tx.contractAddress || tx.to) )]; status.innerText = `Found ${contracts.length} deployments. Loading markets...`; // 3. Loop with a "Breath" (100ms delay) so we don't crash the RPC for (const addr of contracts) { try { const contract = new ethers.Contract(addr, ABI, provider); const count = await contract.marketCount(); for (let i = 0; i < Number(count); i++) { const m = await contract.getMarket(i); addCardToGrid(addr, i, m); } // Small delay to prevent "429 Too Many Requests" await new Promise(resolve => setTimeout(resolve, 100)); } catch (e) { console.log(`Skipping: ${addr} (Not a compatible market)`); } } status.innerText = `✅ Loaded ${grid.children.length} markets from ${contracts.length} deployments.`; } catch (err) { status.innerText = "❌ Connection Error. Try refreshing in 10 seconds."; console.error(err); } }
